\subsection{git clone}
\begin{frame}[fragile]
  \subslidetitle
  The command \cmd{git clone [option(s)] <repository> [<directory>]} is used to download a repository's whole history.
  \begin{lstlisting}
git clone https://github.com/git/git.git
  \end{lstlisting}

% protocoles
git protocoles:
\begin{itemize}
\item http[s]
\item ssh
\item git
\item ftp rsync ...
\end{itemize}
\end{frame}
